select kc.name as company_name, km.bank_name as mechanism_name, ke.name as edition_name, kba.* from kw_bank_account kba left join kw_edition ke on ke.id = kba.edition_id left join kw_company kc on kba.relation_type = 1 and kba.relation_id = kc.id left join kw_mechanism km on km.id = ke.mechanism_id where kba.deleted=0 order by kba.when_created desc
